const cases = {};

cases['ERRORCODEHERE'] = function () {};

function outputError(error) {
  let errorMessage;
  if (typeof cases[error] == 'function') {
    // only executes if we've defined it above
    errorMessage = cases[error]();
  } else {
    // default (the fallthrough)
  }

  return errorMessage;
}

export default outputError;
